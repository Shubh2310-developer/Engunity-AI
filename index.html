Document Processing Service

class DocumentProcessor:
    def __init__(self):
        self.embedding_model = SentenceTransformer('all-MiniLM-L6-v2')
        self.faiss_index = faiss.IndexFlatIP(384)
        self.chunk_size = 512
        self.chunk_overlap = 64
    
    async def process_document(self, file_path: str, user_id: str):
        # 1. Extract text based on file type
        text = await self.extract_text(file_path)
        
        # 2. Intelligent chunking
        chunks = await self.smart_chunk_text(text)
        
        # 3. Generate embeddings
        embeddings = self.embedding_model.encode(chunks)
        
        # 4. Store in FAISS + metadata in DB
        doc_id = await self.store_embeddings(embeddings, chunks, user_id)
        
        return doc_id

RAG Query Service

class RAGQueryService:
    def __init__(self):
        self.retriever = FaissRetriever()
        self.reranker = CrossEncoder('cross-encoder/ms-marco-MiniLM-L-12-v2')
        self.groq_client = GroqClient()
        self.local_llm = LocalPhi2Model()
    
    async def answer_query(self, query: str, doc_id: str, user_id: str):
        # 1. Process query
        processed_query = await self.preprocess_query(query)
        
        # 2. Retrieve relevant chunks
        candidates = await self.retriever.search(processed_query, doc_id, k=10)
        
        # 3. Rerank for relevance
        reranked = await self.rerank_results(query, candidates)
        
        # 4. Assemble context
        context = self.assemble_context(reranked[:5])
        
        # 5. Generate answer with fallback
        try:
            answer = await self.groq_client.generate(query, context)
        except Exception:
            answer = await self.local_llm.generate(query, context)
        
        # 6. Cache and return
        await self.cache_response(query, answer, doc_id)
        return answer
